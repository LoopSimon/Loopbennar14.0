namespace = deforestation

country_event = {
	id = deforestation.1
	title = deforestation.1.t
	desc = deforestation.1.d
	picture = FALUN_COPPER_MINE_eventPicture

	is_triggered_only = no
	trigger = {
		any_owned_province = {
			had_province_flag = {
				flag = deforestation_1
				days = 1095
			}
		}
	}

	mean_time_to_happen = { days = 91 }

	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = { 
					had_province_flag = {
						flag = deforestation_1
						days = 1095
					}
				}
				save_event_target_as = deforestation_event
				add_human_minority_size_effect = yes
				clr_province_flag = deforestation_1

				if = {
					limit = {
						trade_goods = fur
					}
					random_list = {
						20 = { change_trade_goods = grain }
						20 = { change_trade_goods = livestock }
						20 = { change_trade_goods = ivory }
						8 = { change_trade_goods = naval_supplies }
						8 = { change_trade_goods = copper }
						8 = { change_trade_goods = iron }
						5 = { change_trade_goods = gold }
						5 = { change_trade_goods = gems }
						2 = {
							if = {
								limit = { is_year = 1700 }
								change_trade_goods = coal
							}
							else = {
								change_trade_goods = gold
							}
						}
						2 = {
							change_trade_goods = damestear
							add_permanent_province_modifier = {
								name = damestear_rich_deposit
								duration = -1
							}
						}
						2 = {
							change_trade_goods = precursor_relics
						}
					}
				}
			}
		}
	}

	option = {
		name = deforestation.1.a
		ai_chance = { factor = 100 }
		goto = deforestation_event
		tooltip = {
			event_target:deforestation_event = {
				custom_tooltip = bayvic_deforestation_event_tt
				add_human_minority_size_effect = yes
			}
		}
	}
}
